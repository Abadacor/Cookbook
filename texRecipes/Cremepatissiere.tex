    \begin{minipage}{.7\textwidth}
        \begin{enumerate}
            \item Mélanger l'oeuf avec le sucre et la farine
	    \item Délayer dans le lait bouillant
	    \item Cuire jusqu'à épaississement
	    \item Hors du feu, ajouter le beurre et le rhum.

        \end{enumerate}
    \end{minipage}
    \begin{minipage}{.3\textwidth}
        \begin{flushleft}
        \begin{itemize}
            \item 1 oeuf
	    \item 60g de sucre
	    \item 30g de farine
	    \item 250mL de lait
	    \item 30g de beurre
	    \item un bouchon de rhum

        \end{itemize}
